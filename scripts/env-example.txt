# Exemplo de variáveis de ambiente para novo projeto Supabase
# Copie este arquivo para .env.local na raiz do projeto e preencha com seus dados

# Supabase Configuration
VITE_SUPABASE_URL=https://seu-project-id.supabase.co
VITE_SUPABASE_ANON_KEY=sua-anon-key-aqui

# Opcional: Para desenvolvimento local
# VITE_SUPABASE_URL=http://localhost:54321
# VITE_SUPABASE_ANON_KEY=sua-local-anon-key

# Como obter essas informações:
# 1. Acesse https://supabase.com/dashboard
# 2. Selecione seu projeto
# 3. Vá em Settings > API
# 4. Copie a Project URL e a anon/public key
